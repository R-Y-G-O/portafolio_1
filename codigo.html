// Filtro dinámico de propiedades — IM.FILIO
function filtrarTarjetas() {
  const ciudad    = document.getElementById('ciudad').value;
  const precioMax = Number(document.getElementById('precio').value);
  const areaMin   = Number(document.getElementById('area').value);
  let coincidencias = 0;

  document.querySelectorAll('.tarjeta_filtro').forEach(tarjeta => {
    const cumpleCiudad = ciudad ? tarjeta.dataset.ciudad === ciudad : true;
    const cumplePrecio = precioMax ? Number(tarjeta.dataset.precio) <= precioMax : true;
    const cumpleArea   = areaMin  ? Number(tarjeta.dataset.area)   >= areaMin   : true;

    const visible = cumpleCiudad && cumplePrecio && cumpleArea;
    tarjeta.style.display = visible ? 'block' : 'none';
    if (visible) coincidencias++;
  });

  document.getElementById('mensaje-no-coincidencias')
    .style.display = coincidencias === 0 ? 'block' : 'none';
}

// Contador animado con IntersectionObserver — GMnorte
function startCounting() {
  document.querySelectorAll('.counter').forEach(counter => {
    counter.innerText = '0';
    const target    = +counter.getAttribute('data-target');
    const increment = target / 100;

    const update = () => {
      const current = +counter.innerText;
      if (current < target) {
        counter.innerText = Math.ceil(current + increment);
        setTimeout(update, 30);
      } else {
        counter.innerText = target;
      }
    };
    update();
  });
}

// Solo se activa cuando la sección es visible
const observer = new IntersectionObserver(entries => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      startCounting();
      observer.unobserve(entry.target);
    }
  });
}, { threshold: 0.5 });

observer.observe(document.querySelector('.stats-section'));

// 1. CONTROLADOR DELGADO (ProjectController.php)
namespace App\Http\Controllers;

use App\Http\Requests\StoreProjectRequest;
use App\Services\ProjectService;
use Illuminate\Http\JsonResponse;

class ProjectController extends Controller
{
    public function __construct(protected ProjectService $projectService) {}

    public function store(StoreProjectRequest $request): JsonResponse
    {
        // El FormRequest ya validó y sanitizó los datos automáticamente
        $project = $this->projectService->createProject($request->validated());

        return response()->json([
            'message' => 'Proyecto creado y procesado correctamente.',
            'data' => $project
        ], 201);
    }
}

// 2. CAPA DE SERVICIO DESACOPLADA (ProjectService.php)
namespace App\Services;

use App\Models\Project;
use App\Jobs\OptimizeProjectImages;
use Illuminate\Support\Facades\DB;

class ProjectService
{
    public function createProject(array $data): Project
    {
        // Garantizamos atomicidad usando transacciones de Base de Datos
        return DB::transaction(function () use ($data) {
            $project = Project::create([
                'title' => $data['title'],
                'description' => $data['description'],
                'client_id' => $data['client_id'],
            ]);

            if (!empty($data['technologies'])) {
                $project->technologies()->sync($data['technologies']);
            }

            // Delegamos procesos pesados a la cola (Queue) con un Job
            OptimizeProjectImages::dispatch($project);

            return $project;
        });
    }

    public function getActiveProjects()
    {
        // Rendimiento: Eager loading (with) para prevenir el problema de queries N+1
        return Project::with(['technologies', 'client'])
            ->where('status', 'active')
            ->latest()
            ->paginate(12);
    }
}

// 3. VALIDACIÓN Y SANITIZACIÓN SEGURA (StoreProjectRequest.php)
namespace App\Http\Requests;

use Illuminate\Foundation\Http\FormRequest;

class StoreProjectRequest extends FormRequest
{
    public function authorize(): bool
    {
        // Autorización delegada mediante Laravel Policies
        return $this->user()->can('create', Project::class);
    }

    public function rules(): array
    {
        return [
            'title'        => 'required|string|max:100|unique:projects,title',
            'description'  => 'required|string|min:10',
            'client_id'    => 'required|exists:clients,id',
            'technologies' => 'array|exists:technologies,id',
        ];
    }
}

function buildLineChart(range = '6m') {
  const ctx  = document.getElementById('lineChart').getContext('2d');
  const data = trafficData[range];

  if (lineChart) lineChart.destroy();

  lineChart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: data.labels,
      datasets: [{
        label: 'Visitantes',
        data: data.visitors,
        borderColor: '#6366f1',
        backgroundColor: 'rgba(99,102,241,0.1)',
        borderWidth: 2.5,
        tension: 0.4,
        fill: true,
      }],
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      interaction: { mode: 'index', intersect: false },
      plugins: { legend: { position: 'top' } },
      scales: {
        y: {
          ticks: {
            callback: v => v >= 1000 ? (v / 1000).toFixed(1) + 'k' : v,
          },
        },
      },
    },
  });
}
