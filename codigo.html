// Filtro dinámico de propiedades — IM.FILIO
function filtrarTarjetas() {
  const ciudad    = document.getElementById('ciudad').value;
  const precioMax = Number(document.getElementById('precio').value);
  const areaMin   = Number(document.getElementById('area').value);
  let coincidencias = 0;

  document.querySelectorAll('.tarjeta_filtro').forEach(tarjeta => {
    const cumpleCiudad = ciudad ? tarjeta.dataset.ciudad === ciudad : true;
    const cumplePrecio = precioMax ? Number(tarjeta.dataset.precio) <= precioMax : true;
    const cumpleArea   = areaMin  ? Number(tarjeta.dataset.area)   >= areaMin   : true;

    const visible = cumpleCiudad && cumplePrecio && cumpleArea;
    tarjeta.style.display = visible ? 'block' : 'none';
    if (visible) coincidencias++;
  });

  document.getElementById('mensaje-no-coincidencias')
    .style.display = coincidencias === 0 ? 'block' : 'none';
}

// Contador animado con IntersectionObserver — GMnorte
function startCounting() {
  document.querySelectorAll('.counter').forEach(counter => {
    counter.innerText = '0';
    const target    = +counter.getAttribute('data-target');
    const increment = target / 100;

    const update = () => {
      const current = +counter.innerText;
      if (current < target) {
        counter.innerText = Math.ceil(current + increment);
        setTimeout(update, 30);
      } else {
        counter.innerText = target;
      }
    };
    update();
  });
}

// Solo se activa cuando la sección es visible
const observer = new IntersectionObserver(entries => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      startCounting();
      observer.unobserve(entry.target);
    }
  });
}, { threshold: 0.5 });

observer.observe(document.querySelector('.stats-section'));

// Gráfica de líneas con Chart.js — Dashboard Analytics
function buildLineChart(range = '6m') {
  const ctx  = document.getElementById('lineChart').getContext('2d');
  const data = trafficData[range];

  if (lineChart) lineChart.destroy();

  lineChart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: data.labels,
      datasets: [{
        label: 'Visitantes',
        data: data.visitors,
        borderColor: '#6366f1',
        backgroundColor: 'rgba(99,102,241,0.1)',
        borderWidth: 2.5,
        tension: 0.4,
        fill: true,
      }],
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      interaction: { mode: 'index', intersect: false },
      plugins: { legend: { position: 'top' } },
      scales: {
        y: {
          ticks: {
            callback: v => v >= 1000 ? (v / 1000).toFixed(1) + 'k' : v,
          },
        },
      },
    },
  });
}

// Cambiar rango al hacer clic en los botones
document.querySelectorAll('.chart-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.chart-btn')
      .forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    buildLineChart(btn.dataset.range);
  });
});
